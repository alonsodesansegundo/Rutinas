Implementation

Future<void> deletePartidaById(int partidaId, [Database? db]) async {
  final Database database = db ?? await initializeDB();

  // Borrar la partida de la tabla partida
  await database.delete(
    'partida',
    where: 'id = ?',
    whereArgs: [partidaId],
  );
}
