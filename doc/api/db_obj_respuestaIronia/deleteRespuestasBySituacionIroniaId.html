Future<void> deleteRespuestasBySituacionIroniaId(
    Database db, int situacionIroniaId) async {
  try {
    await db.delete(
      'respuestaIronia',
      where: 'situacionIroniaId = ?',
      whereArgs: [situacionIroniaId],
    );
  } catch (e) {
    print("Error al eliminar respuestas: $e");
  }
}
