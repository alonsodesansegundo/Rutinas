void deleteAccion(Database database, int accionId) async {
  try {
    await database.delete(
      'accion',
      where: 'id = ?',
      whereArgs: [accionId],
    );
  } catch (e) {
    print('Error al borrar la instancia de accion: $e');
  }
}
